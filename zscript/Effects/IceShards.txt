Class IceShard1 : Actor
{
	override void BeginPlay(void)
	{
		ChangeStatNum(STAT_PB_BULLETS);
		NashGoreStatics.QueueCasings();
		Super.BeginPlay();
	}
   Default{
    Radius 4;
    Height 4;
    Speed 8;
    Scale 0.6;	
    Mass 1;
    +NOBLOCKMAP;
	+MISSILE;
    +NOTELEPORT;
    +MOVEWITHSECTOR;
    +CLIENTSIDEONLY;
	-DONTSPLASH;
	+THRUGHOST;
	+THRUACTORS;
	BounceType "Doom";
	Alpha 1;
	BounceFactor 0.4;
	Mass 1;
   SeeSound "IceShardBounce";
   }
   States
   {
		Spawn:
		TNT1 A 1;
		TNT1 A 0 A_JumpIf(waterlevel > 1, "Death");
		TNT1 A 0 A_Jump(256,"SpinFast","SpinMedium","SpinSlow");
		SpinFast:
				CSC1 A 100 {
				A_SetRoll(roll+frandom(-80,-120));
					}
		SpinMedium:
				CSC1 A 100 {
				A_SetRoll(roll+frandom(-25,-90));
					}
		SpinSlow:
				CSC1 A 100 {
				A_SetRoll(roll+36);
					}
		Goto Death;
		
		Death:
		  TNT1 I 0 A_SetRoll(0);
		  TNT1 A 0 A_Jump(256,"Land1","Land2","Land3","Land4");
		Land1:
			CSC1 A 1;
			Goto Rest;
		Land2:
			CSC1 B 1;
			Goto Rest;
		Land3:
			CSC1 C 1;
			Goto Rest;
		Land4:
			CSC1 D 1;
			Goto Rest;
		Rest:
			"####" "###########################################" 2;
			"####" "#" -1;
			Stop;
   }
}
