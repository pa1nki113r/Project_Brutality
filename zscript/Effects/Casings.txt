	bool smoking;

		NashGoreStatics.QueueCasings();
		Super.BeginPlay();
	}
   Default
   {Height 2;
   Radius 2;
   Speed 8;
   Scale 0.1;
   +DOOMBOUNCE;
   - NOGRAVITY;
   +WINDTHRUST;

   +MOVEWITHSECTOR;
   +MISSILE;

   -DROPOFF;
   +NOTELEPORT;

   +NOTDMATCH;
   +GHOST;
   -DONTSPLASH;
   -CANBOUNCEWATER;

   +THRUACTORS;
   +ROLLSPRITE;
   Mass 4;
   SeeSound "weapons/casing";
   }
   States
   {
		Spawn:
			TNT1 A 0;
			TNT1 A 0 A_Jump(16, "SpawnSmoking");
			CAS3 ABCDEFGHIJABCDEFGHIJABCDEFGHIJABCDEFGHIJABCDEFGHIJABCDEFGHIJ 2 {
			A_SetRoll(roll+36);
		}
		Goto Death;
		SpawnSmoking:
			TNT1 A 0 { smoking = 1;}
			CAS3 ABCDEFGHIJABCDEFGHIJABCDEFGHIJABCDEFGHIJABCDEFGHIJABCDEFGHIJ 2 {
			A_SpawnItemEx("CasingSmoke",frandom(0.3,0.2),random(0,-1.0),0,0,0,frandom(0.5,0.1),0,SXF_CLIENTSIDE,0);
			A_SetRoll(roll+36);
		}
		Goto Death;
		
		Death:
		  CAS3 I 0 A_SetRoll(0);
		  LCPJ A 0 A_Jump(256,"Case1","Case2","Case3","Case4","Case5","Case6","Case7","Case8","Case9","Case10");
		Case1:
			CAS3 K 1;
			Goto Rest;
		Case2:
			CAS3 L 1;
			Goto Rest;
		Case3:
			CAS3 M 1;
			Goto Rest;
		Case4:
			CAS3 N 1;
			Goto Rest;
		Case5:
			CAS3 O 1;
			Goto Rest;
		Case6:
			CAS3 P 1;
			Goto Rest;
		Case7:
			CAS3 Q 1;
			Goto Rest;
		Case8:
			CAS3 R 1;
			Goto Rest;
		Case9:
			CAS3 S 1;
			Goto Rest;
		Case10:
			CAS3 T 1;
			Goto Rest;
		Rest:
			"####" "#" 0 A_JumpIf(smoking == 1, "Rest2");
			"####" "###########################################" 2;
			"####" "#" -1;
			Stop;
		Rest2:
			"####" "###########################################" 2 A_SpawnItemEx("CasingSmoke",frandom(0.3,0.2),random(0,-1.0),0,0,0,frandom(0.5,0.1),0,SXF_CLIENTSIDE,0);
			"####" "#" -1;
			Stop;
   
		Splash:
			TNT1 A 0;
			TNT1 A 0 A_SpawnItem("UnderwaterEmptyBrass");
			Stop;
	
   }
}
