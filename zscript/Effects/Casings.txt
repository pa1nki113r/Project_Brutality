Class EmptyBrass : Actor
{
		override void BeginPlay(void)
	{
		ChangeStatNum(STAT_PB_BULLETS);
		NashGoreStatics.QueueCasings();
		Super.BeginPlay();
	}
   Default
   {Height 2;
   Radius 2;
   Speed 8;
   Scale 0.15;
   +DOOMBOUNCE;
   - NOGRAVITY;
   +WINDTHRUST;
   +CLIENTSIDEONLY;
   +MOVEWITHSECTOR;
   +MISSILE;
   +NOBLOCKMAP;
   -DROPOFF;
   +NOTELEPORT;
   +FORCEXYBILLBOARD;
   +NOTDMATCH;
   +GHOST;
   -DONTSPLASH;
   -CANBOUNCEWATER;
   +FLOORCLIP;
   +THRUACTORS;
   +ROLLSPRITE;
   Mass 4;
   SeeSound "weapons/casing";
   }
   States
   {
   Spawn:
	  CAS3 A 1 ;
	  TNT1 A 0 ACS_ExecuteAlways(7798, 0, 0, 0, 0);//Check Effects
	Exist:
       TNT1 A 0 A_SetRoll(roll + (30),SPF_INTERPOLATE);
	   CAS3 G 1 ;
	  //CAS3 ABCDEFGHABCDEFGHABCDEFGHABCDEFGHABCDEFGHABCDEFGH 3 A_JumpIf(waterlevel > 1, "Splash");
	RotateRight:
	  TNT1 A 0 A_SetRoll(roll + (10),SPF_INTERPOLATE);
	  CAS3 G 1 A_JumpIf(waterlevel > 1, "Splash");
	  Loop;

   Death:
      TNT1 A 0 A_SetRoll(0,SPF_INTERPOLATE);
	  CAS3 I 0;
      LCPJ A 0 A_Jump(256,"Rest1","Rest2","Rest3","Rest4","Rest5","Rest6","Rest7","Rest8");
      Goto Rest1;
    Rest1:
      CAS3 I 900;
	  TNT1 A 0 A_JumpIfInventory("CasingsJanitor", 1, "Disappear");
      Loop;
    Rest2:
      CAS3 J 900;
	  TNT1 A 0 A_JumpIfInventory("CasingsJanitor", 1, "Disappear");
      loop;
    Rest3:
	  CAS3 K 900;
	  TNT1 A 0 A_JumpIfInventory("CasingsJanitor", 1, "Disappear");
      Loop;
    Rest4:
      CAS3 L 900;
	  TNT1 A 0 A_JumpIfInventory("CasingsJanitor", 1, "Disappear");
      Loop;
    Rest5:
      CAS3 M 900;
	  TNT1 A 0 A_JumpIfInventory("CasingsJanitor", 1, "Disappear");
      Loop;
    Rest6:
      CAS3 I 900;
	  TNT1 A 0 A_JumpIfInventory("CasingsJanitor", 1, "Disappear");
      Loop;
    Rest7:
      CAS3 J 900;
	  TNT1 A 0 A_JumpIfInventory("CasingsJanitor", 1, "Disappear");
      Loop;
    Rest8:
      CAS3 K 900;
	  TNT1 A 0 A_JumpIfInventory("CasingsJanitor", 1, "Disappear");
      Loop;
	  
  Disappear:
  TNT1 A 1;
  Stop;
	 
	Splash:
	TNT1 A 0;
	TNT1 A 0 A_SpawnItem("UnderwaterEmptyBrass");
	Stop;
	
   }
}
